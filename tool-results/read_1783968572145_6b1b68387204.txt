   957‚Üí   957‚Üífunction computeHealthScore(inv){
   958‚Üí   958‚Üí  if(!inv.length)return{score:0,label:'No Data',details:[]};
   959‚Üí   959‚Üí  const totalItems=inv.length;
   960‚Üí   960‚Üí  const lowStock=inv.filter(p=>p.current<=p.reorder).length;
   961‚Üí   961‚Üí  const outOfStock=inv.filter(p=>p.current===0).length;
   962‚Üí   962‚Üí  const stockVal=inv.reduce((a,b)=>a+b.current*b.cost,0);
   963‚Üí   963‚Üí  const avgCoverage=inv.reduce((a,b)=>a+(b.current>0?b.current/b.reorder*7:0),0)/totalItems;
   964‚Üí   964‚Üí  const variety=totalItems;
   965‚Üí   965‚Üí  const lowPct=lowStock/totalItems;
   966‚Üí   966‚Üí  const outPct=outOfStock/totalItems;
   967‚Üí   967‚Üí  let score=Math.round(100 - lowPct*30 - outPct*40 - Math.max(0,(7-avgCoverage)/7*20));
   968‚Üí   968‚Üí  score=Math.max(0,Math.min(100,score));
   969‚Üí   969‚Üí  let label=score>=80?'Excellent':score>=60?'Good':score>=40?'Fair':score>=20?'Poor':'Critical';
   970‚Üí   970‚Üí  return {score,label,details:[
   971‚Üí   971‚Üí    {label:'Total SKUs',value:variety,good:true},
   972‚Üí   972‚Üí    {label:'Low stock items',value:lowStock,good:lowStock===0},
   973‚Üí   973‚Üí    {label:'Out of stock',value:outOfStock,good:outOfStock===0},
   974‚Üí   974‚Üí    {label:'Avg coverage (days)',value:avgCoverage.toFixed(1),good:avgCoverage>=7},
   975‚Üí   975‚Üí    {label:'Stock value',value:fmt(stockVal),good:true}
   976‚Üí   976‚Üí  ]};
   977‚Üí   977‚Üí}
