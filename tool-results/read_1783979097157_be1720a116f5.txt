   132→   1. STOCK TAKE PAGE

   134→export function StockTakePage() {
   135→  const { storeId, isManager, user } = useAppStore()
   136→  const [loading, setLoading] = useState(true)
   137→  const [error, setError] = useState('')
   138→  const [stockTakes, setStockTakes] = useState<any[]>([])
   139→  const [activeCount, setActiveCount] = useState<any>(null)
   140→  const [counts, setCounts] = useState<Record<string, number>>({})
   141→  const [search, setSearch] = useState('')
   142→  const [expandedId, setExpandedId] = useState<string | null>(null)
   143→  const [submitting, setSubmitting] = useState(false)
   144→
   145→  const fetchStockTakes = async () => {
   146→    if (!storeId) return
   147→    setLoading(true)
   148→    setError('')
   149→    try {
   150→      const data = await api.getStockTakes('storeId=' + storeId)
   151→      const list = Array.isArray(data) ? data : data.stockTakes || data.items || []
   152→      setStockTakes(list)
   153→      const active = list.find((st: any) => st.status === 'in_progress')
   154→      if (active) {
   155→        setActiveCount(active)
   156→        const initCounts: Record<string, number> = {}
   157→        ;(active.items || []).forEach((item: any) => {
   158→          initCounts[item.productId] = item.counted ?? item.expected ?? 0
   159→        })
   160→        setCounts(initCounts)
   161→      }
   162→    } catch (err: any) {
   163→      setError(err.message || 'Failed to load stock takes')
   164→    } finally {
   165→      setLoading(false)
   166→    }
   167→  }
   168→
   169→  useEffect(() => { fetchStockTakes() }, [storeId])
   170→
   171→  const handleStart = async () => {
   172→    if (!storeId) return
   173→    setSubmitting(true)
   174→    setError('')
   175→    try {
   176→      const data = await api.createStockTake(storeId)
   177→      const newSt = data.stockTake || data
   178→      setActiveCount(newSt)
   179→      setCounts({})
   180→      setStockTakes([newSt, ...stockTakes])
   181→    } catch (err: any) {
   182→      setError(err.message || 'Failed to start stock take')
   183→    } finally {
   184→      setSubmitting(false)
   185→    }
   186→  }
   187→
   188→  const handleSubmit = async () => {
   189→    if (!activeCount) return
   190→    setSubmitting(true)
   191→    setError('')
   192→    try {
   193→      const items = (activeCount.items || []).map((item: any) => ({
   194→        id: item.id,
   195→        productId: item.productId,
   196→        counted: counts[item.productId] ?? item.expected ?? 0,
   197→      }))
   198→      await api.updateStockTake({ stockTakeId: activeCount.id, items, action: 'submit' })
   199→      await fetchStockTakes()
   200→      setActiveCount(null)
   201→      setCounts({})
   202→    } catch (err: any) {
   203→      setError(err.message || 'Failed to submit stock take')
   204→    } finally {
   205→      setSubmitting(false)
   206→    }
   207→  }
   208→
   209→  const handleApprove = async (st: any) => {
   210→    setSubmitting(true)
   211→    setError('')
   212→    try {
   213→      await api.updateStockTake({ stockTakeId: st.id, items: st.items || [], action: 'approve' })
   214→      await fetchStockTakes()
   215→    } catch (err: any) {
   216→      setError(err.message || 'Failed to approve stock take')
   217→    } finally {
   218→      setSubmitting(false)
   219→    }
   220→  }
   221→
   222→  if (!storeId) return <NoStore />
   223→
   224→  if (loading) return <Spinner />
   225→
   226→  const filteredItems = (activeCount?.items || []).filter((item: any) =>
   227→    (item.product?.name || item.productName || '').toLowerCase().includes(search.toLowerCase())
   228→  )
   229→
   230→  const statusBadge = (status: string) => {
   231→    if (status === 'in_progress') return <Badge variant="amber">In Progress</Badge>
   232→    if (status === 'pending') return <Badge variant="blue">Pending</Badge>
   233→    if (status === 'approved') return <Badge variant="green">Approved</Badge>
   234→    return <Badge variant="gray">{status}</Badge>
   235→  }
   236→
   237→  return (
   238→    <div className="space-y-6">
   239→      <div className="flex flex-col sm:flex-row sm:items-center sm:justify-between gap-4">
   240→        <div>
   241→          <h1 className="text-xl font-bold" style={{ color: '#1F1129' }}>Stock Counting</h1>
   242→          <p className="text-sm mt-0.5" style={{ color: '#8B7FA0' }}>Conduct physical stock counts and reconcile with system records</p>
   243→        </div>
   244→        {!activeCount && (
   245→          <button onClick={handleStart} disabled={submitting} className={btnPrimary} style={{ backgroundColor: '#DC2626' }}>
   246→            <svg className="w-4 h-4" viewBox="0 0 20 20" fill="currentColor"><path fillRule="evenodd" d="M10 3a1 1 0 011 1v5h5a1 1 0 110 2h-5v5a1 1 0 11-2 0v-5H4a1 1 0 110-2h5V4a1 1 0 011-1z" clipRule="evenodd" /></svg>
   247→            Start New Count
   248→          </button>
   249→        )}
   250→      </div>
   251→
   252→      {error && <ErrorBanner message={error} />}
   253→
   254→      {/* ── Active Count View ── */}
   255→      {activeCount && (
   256→        <div style={cardStyle} className="p-5">
   257→          <div className="flex items-center justify-between mb-4">
   258→            <h2 className="text-base font-bold" style={{ color: '#1F1129' }}>Active Stock Count</h2>
   259→            <Badge variant="amber">In Progress</Badge>
   260→          </div>
   261→
   262→          <div className="mb-4">
   263→            <input
   264→              type="text"
   265→              placeholder="Search products..."
   266→              value={search}
   267→              onChange={(e) => setSearch(e.target.value)}
   268→              className={inputCls}
   269→              style={{ borderColor: '#E8E0F0' }}
   270→            />
   271→          </div>
   272→
   273→          <div className="overflow-x-auto">
   274→            <table className="w-full text-sm">
   275→              <thead>
   276→                <tr className="border-b border-[#E8E0F0]">
   277→                  <th className="text-left py-2.5 px-3 text-xs font-semibold uppercase tracking-wider" style={{ color: '#8B7FA0' }}>Product</th>
   278→                  <th className="text-right py-2.5 px-3 text-xs font-semibold uppercase tracking-wider" style={{ color: '#8B7FA0' }}>Expected</th>
   279→                  <th className="text-right py-2.5 px-3 text-xs font-semibold uppercase tracking-wider" style={{ color: '#8B7FA0' }}>Counted</th>
   280→                </tr>
   281→              </thead>
   282→              <tbody>
   283→                {filteredItems.map((item: any) => (
   284→                  <tr key={item.productId} className="border-b border-[#F5F0EB] last:border-0">
   285→                    <td className="py-2.5 px-3 font-medium" style={{ color: '#1F1129' }}>{item.product?.name || item.productName || item.name || 'Unknown'}</td>
   286→                    <td className="py-2.5 px-3 text-right" style={{ color: '#6B5E7A' }}>{item.expected ?? 0}</td>
   287→                    <td className="py-2.5 px-3 text-right">
   288→                      <input
   289→                        type="number"
   290→                        min="0"
   291→                        value={counts[item.productId] ?? ''}
   292→                        onChange={(e) => setCounts({ ...counts, [item.productId]: parseInt(e.target.value) || 0 })}
   293→                        className="w-20 text-right px-2 py-1 rounded-lg border text-sm outline-none focus:ring-2 focus:ring-[#DC2626]/20 focus:border-[#DC2626]"
   294→                        style={{ borderColor: '#E8E0F0' }}
   295→                      />
   296→                    </td>
   297→                  </tr>
   298→                ))}
   299→                {filteredItems.length === 0 && (
   300→                  <tr><td colSpan={3} className="py-8 text-center text-sm" style={{ color: '#8B7FA0' }}>No products found</td></tr>
   301→                )}
   302→              </tbody>
   303→            </table>
   304→          </div>
   305→
   306→          <div className="flex justify-end mt-4">
   307→            <button onClick={handleSubmit} disabled={submitting || filteredItems.length === 0} className={btnPrimary} style={{ backgroundColor: '#F59E0B' }}>
   308→              {submitting ? 'Submitting...' : 'Submit for Approval'}
   309→            </button>
   310→          </div>
   311→        </div>
   312→      )}
   313→
   314→      {/* ── Pending notice for staff ── */}
   315→      {!activeCount && stockTakes.some((st: any) => st.status === 'pending') && !isManager && (
   316→        <div style={cardStyle} className="p-5 text-center">
   317→          <svg className="w-10 h-10 mx-auto mb-3" viewBox="0 0 24 24" fill="none" stroke="#3B82F6" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round">
   318→            <circle cx="12" cy="12" r="10" /><path d="M12 6v6l4 2" />
   319→          </svg>
   320→          <h3 className="text-base font-bold mb-1" style={{ color: '#1F1129' }}>Awaiting Manager Approval</h3>
   321→          <p className="text-sm" style={{ color: '#8B7FA0' }}>Your stock count has been submitted and is waiting for a manager to review and approve it.</p>
   322→        </div>
   323→      )}
   324→
   325→      {/* ── History ── */}
   326→      <div style={cardStyle} className="p-5">
   327→        <h2 className="text-base font-bold mb-4" style={{ color: '#1F1129' }}>Stock Count History</h2>
   328→        <div className="space-y-2">
   329→          {stockTakes.map((st: any) => (
   330→            <div key={st.id} className="rounded-xl border border-[#E8E0F0] overflow-hidden">
   331→              <button
   332→                onClick={() => setExpandedId(expandedId === st.id ? null : st.id)}
   333→                className="w-full flex items-center justify-between px-4 py-3 text-left hover:bg-[#FAF7F2] transition cursor-pointer"
   334→              >
   335→                <div className="flex items-center gap-3">
   336→                  {statusBadge(st.status)}
   337→                  <span className="text-sm font-medium" style={{ color: '#1F1129' }}>
   338→                    {st.createdAt ? new Date(st.createdAt).toLocaleDateString('en-KE', { day: '2-digit', month: 'short', year: 'numeric' }) : 'N/A'}
   339→                  </span>
   340→                  <span className="text-xs" style={{ color: '#8B7FA0' }}>
   341→                    {(st.items || []).length} items
   342→                  </span>
   343→                </div>
   344→                <svg className={`w-4 h-4 transition-transform ${expandedId === st.id ? 'rotate-180' : ''}`} viewBox="0 0 20 20" fill="#8B7FA0"><path fillRule="evenodd" d="M5.293 7.293a1 1 0 011.414 0L10 10.586l3.293-3.293a1 1 0 111.414 1.414l-4 4a1 1 0 01-1.414 0l-4-4a1 1 0 010-1.414z" clipRule="evenodd" /></svg>
   345→              </button>
   346→              {expandedId === st.id && (
   347→                <div className="border-t border-[#E8E0F0] px-4 py-3">
   348→                  <table className="w-full text-sm">
   349→                    <thead>
   350→                      <tr className="border-b border-[#F5F0EB]">
   351→                        <th className="text-left py-2 text-xs font-semibold uppercase tracking-wider" style={{ color: '#8B7FA0' }}>Product</th>
   352→                        <th className="text-right py-2 text-xs font-semibold uppercase tracking-wider" style={{ color: '#8B7FA0' }}>Expected</th>
   353→                        <th className="text-right py-2 text-xs font-semibold uppercase tracking-wider" style={{ color: '#8B7FA0' }}>Counted</th>
   354→                        <th className="text-right py-2 text-xs font-semibold uppercase tracking-wider" style={{ color: '#8B7FA0' }}>Variance</th>
   355→                      </tr>
   356→                    </thead>
   357→                    <tbody>
   358→                      {(st.items || []).map((item: any) => {
   359→                        const variance = (item.countedQty ?? 0) - (item.expectedQty ?? 0)
   360→                        return (
   361→                          <tr key={item.productId} className="border-b border-[#F5F0EB] last:border-0">
   362→                            <td className="py-2 font-medium" style={{ color: '#1F1129' }}>{item.productName || 'Unknown'}</td>
   363→                            <td className="py-2 text-right" style={{ color: '#6B5E7A' }}>{item.expectedQty ?? 0}</td>
   364→                            <td className="py-2 text-right" style={{ color: '#6B5E7A' }}>{item.countedQty ?? 0}</td>
   365→                            <td className="py-2 text-right font-semibold" style={{ color: variance === 0 ? '#059669' : variance > 0 ? '#D97706' : '#DC2626' }}>
   366→                              {variance > 0 ? '+' : ''}{variance}
   367→                            </td>
   368→                          </tr>
   369→                        )
   370→                      })}
   371→                    </tbody>
   372→                  </table>
   373→                  {isManager && st.status === 'pending' && (
   374→                    <div className="flex justify-end mt-3 pt-3 border-t border-[#E8E0F0]">
   375→                      <button onClick={() => handleApprove(st)} disabled={submitting} className={btnPrimary} style={{ backgroundColor: '#059669' }}>
   376→                        {submitting ? 'Approving...' : 'Approve Stock Take'}
   377→                      </button>
   378→                    </div>
   379→                  )}
   380→                </div>
   381→              )}
   382→            </div>
   383→          ))}
   384→          {stockTakes.length === 0 && !activeCount && (
   385→            <p className="text-center text-sm py-8" style={{ color: '#8B7FA0' }}>No stock counts have been recorded yet.</p>
   386→          )}
   387→        </div>
   388→      </div>
   389→    </div>
   390→  )
   391→}

   398→  const [loading, setLoading] = useState(true)
   399→  const [error, setError] = useState('')
   400→  const [date, setDate] = useState(() => new Date().toISOString().split('T')[0])
   401→  const [recon, setRecon] = useState<any>(null)

   403→  const [saving, setSaving] = useState(false)
