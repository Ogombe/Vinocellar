   960→  const lowStock=inv.filter(p=>p.current<=p.reorder).length;
   961→  const outOfStock=inv.filter(p=>p.current===0).length;
   962→  const stockVal=inv.reduce((a,b)=>a+b.current*b.cost,0);
   963→  const avgCoverage=inv.reduce((a,b)=>a+(b.current>0?b.current/b.reorder*7:0),0)/totalItems;
   964→  const variety=totalItems;
   965→  const lowPct=lowStock/totalItems;
   966→  const outPct=outOfStock/totalItems;
   967→  let score=Math.round(100 - lowPct*30 - outPct*40 - Math.max(0,(7-avgCoverage)/7*20));
   968→  score=Math.max(0,Math.min(100,score));
   969→  let label=score>=80?'Excellent':score>=60?'Good':score>=40?'Fair':score>=20?'Poor':'Critical';
   970→  return {score,label,details:[
   971→    {label:'Total SKUs',value:variety,good:true},
   972→    {label:'Low stock items',value:lowStock,good:lowStock===0},
   973→    {label:'Out of stock',value:outOfStock,good:outOfStock===0},
   974→    {label:'Avg coverage (days)',value:avgCoverage.toFixed(1),good:avgCoverage>=7},
   975→    {label:'Stock value',value:fmt(stockVal),good:true}
   976→  ]};
   977→}
